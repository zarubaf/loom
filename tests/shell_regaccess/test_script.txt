# Shell register access test script
# Exercises read, write, status, and couple/decouple commands.

status

# Read Shell Version register
read 0x28

# Read Design Hash word 0
read 0x48

# Write step count = 10
write 0x10 0x0a

# Read back step count to verify write
read 0x10

# Step 5 cycles
step 5

# Read cycle counter low
read 0x08

# Verify couple/decouple commands
decouple
couple

# Ensure emulation is stopped
stop

exit
